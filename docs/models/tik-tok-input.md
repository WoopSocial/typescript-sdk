# TikTokInput

## Example Usage

```typescript
import { TikTokInput } from "@woopsocial/typescript-sdk/models";

let value: TikTokInput = {
  platform: "TIKTOK",
  socialAccountId: "<id>",
  postType: "PHOTO",
  privacyLevel: "SELF_ONLY",
  allowComment: true,
  allowDuet: true,
  allowStitch: true,
  contentDisclosureEnabled: false,
  isYourBrand: false,
  isBrandedContent: true,
  autoAddMusic: true,
};
```
